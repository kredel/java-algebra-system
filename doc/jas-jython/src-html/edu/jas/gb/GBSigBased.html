024/**
025 * Groebner bases via signatur based GBs in script.
026 * @author Heinz Kredel
027 * TODO: Computing via the ScriptEngine is way slower than the direct execution 
028 * in the jython interpreter. Check if a different engine is in the path or if it must 
029 * be configured in some special way.
030 */
031
032public class GBSigBased<C extends GcdRingElem<C>> extends GroebnerBaseAbstract<C> {
033
034
035    private static final Logger logger = Logger.getLogger(GBSigBased.class);
036
037
038    private final boolean debug = logger.isDebugEnabled(); //logger.isInfoEnabled();
039
040
041    /**
042     * Scripting engine.
043     */
044    public final ScriptEngine engine;
045
046
047    /**
048     * GBSigBased constructor.
049     */
050    public GBSigBased() {
051        engine = new ScriptEngineManager().getEngineByExtension("py");
052        logger.info("Script engine discovered: " + engine);
053        if ( engine == null ) {
054            logger.error("no script engine found");
055            throw new RuntimeException("no script engine found");
056        }
057        StringBuffer sb = new StringBuffer();
058        sb.append("from jas import PolyRing, ZZ, QQ, arraylist2pylist, pylist2arraylist;\n" );
059        sb.append("from basic_sigbased_gb import sigbased_gb, arris_algorithm, ggv, ggv_first_implementation, f5, f5z;\n" );
060        sb.append("sbgb = sigbased_gb();\n" );
061        sb.append("arri = arris_algorithm();\n" );
062        sb.append("ggv = ggv();\n" );
063        sb.append("ggv1 = ggv_first_implementation();\n" );
064        sb.append("ff5 = f5z();\n" );
065        String ex = sb.toString();
066        if (debug) {
067            logger.info("input for evaluation:\n" + ex);
068        }
069        try { 
070            Object ans = engine.eval(ex);
071        } catch (ScriptException e) {
072            e.printStackTrace();
073            throw new RuntimeException(e);
074        }
075    }
076
077
078    /**
079     * Get the String representation with GB engine.
080     * @see java.lang.Object#toString()
081     */
082    @Override
083    public String toString() {
084        return "GBSigBased[ " + engine.getClass().getName() + " ]";
085    }
086
087
088    /**
089     * Cleanup and terminate ThreadPool.
090     */
091    @Override
092    public void terminate() {
093    }
094
095
096    /**
097     * Cancel ThreadPool.
098     */
099    @Override
100    public int cancel() {
101        return 0;
102    }
103
104
105    /**
106     * Groebner base.
107     * @param modv module variable number.
108     * @param F polynomial list.
109     * @return GB(F) a Groebner base of F.
110     */
111    //JAVA6only: @Override
112    public List<GenPolynomial<C>> GB(int modv, List<GenPolynomial<C>> F) {
113        if (F == null || F.isEmpty()) {
114            return F;
115        }
116        if (modv != 0) {
117            throw new UnsupportedOperationException("implemented only for modv = 0, not " + modv);
118        }
119        GenPolynomialRing<C> pfac = F.get(0).ring;
120        List<GenPolynomial<C>> G = new ArrayList<GenPolynomial<C>>();
121        long millis = System.currentTimeMillis();
122        try {
123            engine.put("F",F);
124            StringBuffer sb = new StringBuffer();
125            //sb.append("r = " + pfac.toScript() + ";\n");
126            //sb.append("print str(r);\n");
127            //sb.append("print \"F = \" + str(F);\n");
128            sb.append("Fp = arraylist2pylist(F);\n");
129            //sb.append("Gp = sbgb.basis_sig(Fp);\n");
130            //sb.append("Gp = ff5.basis_sig(Fp);\n");
131            //sb.append("Gp = arri.basis_sig(Fp);\n");
132            sb.append("Gp = ggv1.basis_sig(Fp);\n");
133            sb.append("G = pylist2arraylist(Gp);\n");
134            String ex = sb.toString();
135            if (debug) {
136                logger.info("input for evaluation:\n" + ex);
137            }
138            Object ans = engine.eval(ex);
139            if ( ans != null ) {
140                logger.info("answer: " + ans);
141            }
142            G = (List<GenPolynomial<C>>) engine.get("G");
143        } catch (ScriptException e) {
144            e.printStackTrace();
145        } catch (Exception e) {
146            e.printStackTrace();
147        }
148        millis = System.currentTimeMillis() - millis;
149        System.out.println("evaluation took " + millis);
150        if (debug) {
151            logger.info("polynomials G: " + G);
152        }
153        return G;
154    }
